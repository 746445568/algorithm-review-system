func (db *DB) GetGoals() ([]models.Goal, error) {
    rows, err := db.conn.Query(`SELECT id, platform, title, target_rating, deadline, created_at FROM goals ORDER BY created_at DESC`)
    if err != nil { return nil, err }
    defer rows.Close()
    var goals []models.Goal
    for rows.Next() {
        var g models.Goal
        var deadline sql.NullString
        var createdAt string
        if err := rows.Scan(&g.ID, &g.Platform, &g.Title, &g.TargetRating, &deadline, &createdAt); err != nil {
            return nil, err
        }
        if deadline.Valid {
            t, _ := time.Parse(time.RFC3339, deadline.String)
            g.Deadline = &t
        }
        g.CreatedAt, _ = time.Parse(time.RFC3339, createdAt)
        goals = append(goals, g)
    }
    return goals, rows.Err()
}

func (db *DB) CreateGoal(g models.Goal) (models.Goal, error) {
    db.writeMu.Lock()
    defer db.writeMu.Unlock()
    var deadlineStr interface{}
    if g.Deadline != nil {
        deadlineStr = g.Deadline.UTC().Format(time.RFC3339)
    }
    res, err := db.conn.Exec(
        `INSERT INTO goals(platform, title, target_rating, deadline) VALUES(?,?,?,?)`,
        g.Platform, g.Title, g.TargetRating, deadlineStr,
    )
    if err != nil { return g, err }
    id, _ := res.LastInsertId()
    g.ID = id
    g.CreatedAt = time.Now().UTC()
    return g, nil
}

func (db *DB) DeleteGoal(id int64) error {
    db.writeMu.Lock()
    defer db.writeMu.Unlock()
    _, err := db.conn.Exec(`DELETE FROM goals WHERE id = ?`, id)
    return err
}
```

func (a *CodeforcesAdapter) FetchProfile(handle string) (judges.UserProfile, error) {
    var users []struct {
        Rating    int    `json:"rating"`
        MaxRating int    `json:"maxRating"`
        Rank      string `json:"rank"`
    }
    if err := a.getJSON("user.info", url.Values{"handles": []string{handle}}, &users); err != nil {
        return judges.UserProfile{}, fmt.Errorf("fetch cf profile %q: %w", handle, err)
    }
    if len(users) == 0 {
        return judges.UserProfile{}, fmt.Errorf("cf user not found: %s", handle)
    }
    u := users[0]
    return judges.UserProfile{
        Rating:    &u.Rating,
        MaxRating: &u.MaxRating,
        Rank:      u.Rank,
    }, nil
}

// atcoder.go
func (a *AtCoderAdapter) FetchProfile(handle string) (judges.UserProfile, error) {
    url := fmt.Sprintf("https://kenkoooo.com/atcoder/atcoder-api/v3/user/submissions?user=%s&from_second=0&limit=1", handle)
    // kenkoooo 没有直接的 rating 端点，用 /resources/ac_difficulty.json 方式太大
    // 改用 https://atcoder.jp/users/{handle}/history/json（非官方但稳定）
    histURL := fmt.Sprintf("https://atcoder.jp/users/%s/history/json", handle)
    req, err := http.NewRequest(http.MethodGet, histURL, nil)
    if err != nil { return judges.UserProfile{}, err }
    req.Header.Set("Accept", "application/json")
    resp, err := http.DefaultClient.Do(req)
    if err != nil { return judges.UserProfile{}, fmt.Errorf("fetch atcoder profile: %w", err) }
    defer resp.Body.Close()
    var history []struct {
        NewRating int `json:"NewRating"`
    }
    if err := json.NewDecoder(resp.Body).Decode(&history); err != nil {
        return judges.UserProfile{}, fmt.Errorf("decode atcoder history: %w", err)
    }
    if len(history) == 0 {
        return judges.UserProfile{}, nil
    }
    // 最新分是最后一条
    latest := history[len(history)-1].NewRating
    maxR := 0
    for _, h := range history {
        if h.NewRating > maxR { maxR = h.NewRating }
    }
    return judges.UserProfile{Rating: &latest, MaxRating: &maxR}, nil
}
```

func (s *Server) handleRefreshRating(w http.ResponseWriter, r *http.Request) {
    idStr := r.PathValue("id")
    id, err := strconv.ParseInt(idStr, 10, 64)
    if err != nil {
        writeError(w, http.StatusBadRequest, "invalid account id")
        return
    }
    accounts, err := s.db.GetAccounts()
    if err != nil { writeError(w, http.StatusInternalServerError, err.Error()); return }
    var account *models.PlatformAccount
    for i := range accounts {
        if accounts[i].ID == id { account = &accounts[i]; break }
    }
    if account == nil { writeError(w, http.StatusNotFound, "account not found"); return }
    adapter, ok := s.adapters[account.Platform]
    if !ok { writeError(w, http.StatusBadRequest, "unsupported platform"); return }
    profile, err := adapter.FetchProfile(account.ExternalHandle)
    if err != nil { writeError(w, http.StatusInternalServerError, err.Error()); return }
    if err := s.db.UpdateAccountRating(id, profile.Rating, profile.MaxRating); err != nil {
        writeError(w, http.StatusInternalServerError, err.Error())
        return
    }
    writeJSON(w, http.StatusOK, map[string]any{"rating": profile.Rating, "maxRating": profile.MaxRating})
}

func (s *Server) handleCreateGoal(w http.ResponseWriter, r *http.Request) {
    var input struct {
        Platform     string  `json:"platform"`
        Title        string  `json:"title"`
        TargetRating int     `json:"targetRating"`
        Deadline     *string `json:"deadline"`
    }
    if err := json.NewDecoder(r.Body).Decode(&input); err != nil {
        writeError(w, http.StatusBadRequest, "invalid request body")
        return
    }
    if input.Title == "" || input.TargetRating <= 0 {
        writeError(w, http.StatusBadRequest, "title and targetRating are required")
        return
    }
    g := models.Goal{
        Platform:     models.Platform(input.Platform),
        Title:        input.Title,
        TargetRating: input.TargetRating,
    }
    if input.Deadline != nil {
        t, err := time.Parse(time.RFC3339, *input.Deadline)
        if err == nil { g.Deadline = &t }
    }
    created, err := s.db.CreateGoal(g)
    if err != nil { writeError(w, http.StatusInternalServerError, err.Error()); return }
    writeJSON(w, http.StatusOK, created)
}

// 按周统计提交数和 AC 数（近 12 周）
func (db *DB) GetSubmissionStatsByWeek(weeks int) ([]map[string]any, error) {
    rows, err := db.conn.Query(`
        SELECT
            strftime('%Y-W%W', submitted_at) AS week,
            COUNT(*) AS total,
            SUM(CASE WHEN verdict = 'AC' THEN 1 ELSE 0 END) AS ac_count
        FROM submissions
        WHERE submitted_at >= date('now', ? || ' days')
        GROUP BY week
        ORDER BY week ASC
    `, fmt.Sprintf("-%d", weeks*7))
    if err != nil { return nil, err }
    defer rows.Close()
    var result []map[string]any
    for rows.Next() {
        var week string
        var total, acCount int
        if err := rows.Scan(&week, &total, &acCount); err != nil { return nil, err }
        result = append(result, map[string]any{"week": week, "total": total, "acCount": acCount})
    }
    return result, rows.Err()
}

// 按标签统计正确率
func (db *DB) GetTagAccuracyStats() ([]map[string]any, error) {
    rows, err := db.conn.Query(`
        SELECT
            pt.tag,
            COUNT(DISTINCT s.id) AS attempts,
            SUM(CASE WHEN s.verdict = 'AC' THEN 1 ELSE 0 END) AS ac_count
        FROM submissions s
        JOIN problem_tags pt ON pt.problem_id = s.problem_id
        GROUP BY pt.tag
        HAVING attempts >= 2
        ORDER BY (CAST(ac_count AS REAL)/attempts) ASC
        LIMIT 15
    `)
    if err != nil { return nil, err }
    defer rows.Close()
    var result []map[string]any
    for rows.Next() {
        var tag string
        var attempts, acCount int
        if err := rows.Scan(&tag, &attempts, &acCount); err != nil { return nil, err }
        result = append(result, map[string]any{"tag": tag, "attempts": attempts, "acCount": acCount})
    }
    return result, rows.Err()
}

// 近 90 天每日复习打卡次数
func (db *DB) GetDailyReviewCounts() ([]map[string]any, error) {
    rows, err := db.conn.Query(`
        SELECT
            date(last_updated_at) AS day,
            COUNT(*) AS count
        FROM problem_review_states
        WHERE last_updated_at >= date('now', '-90 days')
        GROUP BY day
        ORDER BY day ASC
    `)
    if err != nil { return nil, err }
    defer rows.Close()
    var result []map[string]any
    for rows.Next() {
        var day string
        var count int
        if err := rows.Scan(&day, &count); err != nil { return nil, err }
        result = append(result, map[string]any{"day": day, "count": count})
    }
    return result, rows.Err()
}

function groupByWeek(contests) {
  const groups = new Map();
  for (const c of contests) {
    const d = new Date(c.startTime);
    // 取周一作为分组键
    const day = d.getDay();
    const monday = new Date(d);
    monday.setDate(d.getDate() - ((day + 6) % 7));
    const key = monday.toISOString().slice(0, 10);

    groups.get(key).push(c);

  return [...groups.entries()].sort(([a], [b]) => a.localeCompare(b));

  const [contests, setContests] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState("");
  const [syncing, setSyncing] = useState(false);
  const [filter, setFilter] = useState("UPCOMING"); // UPCOMING | ALL

  const refresh = useCallback(async () => {
    if (serviceStatus.state !== "healthy") { setLoading(false); return; }
    setLoading(true);
    setError("");
    try {
      const data = await api.getContests({ status: filter === "ALL" ? "" : filter });
      setContests(Array.isArray(data) ? data : []);
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, [serviceStatus.state, filter]);

  useEffect(() => { void refresh(); }, [refresh]);

  async function handleSync() {
    setSyncing(true);
    setError("");
    try {
      await api.syncContests();
      await refresh();
    } catch (e) {
      setError(e.message);
    } finally {
      setSyncing(false);
    }
  }

  const grouped = groupByWeek(contests);

  return (
    <div className="page-grid">
      <section className="panel full-span">
        <div className="panel-header">
          <h3>比赛日历</h3>

              onChange={(e) => setFilter(e.target.value)}
              className="cal-filter-select"
            >
              <option value="UPCOMING">即将开始</option>
              <option value="ALL">全部</option>
            </select>
            <button
              type="button"
              className="ghost-button"
              disabled={syncing || serviceStatus.state !== "healthy"}
              onClick={handleSync}
            >
              {syncing ? "同步中..." : "同步比赛"}
            </button>
          </div>
        </div>

        {loading ? <p className="muted">加载中...</p> : null}

              {weekStart} 周
            </h4>

                const start = new Date(c.startTime);
                const statusClass =
                  c.status === "RUNNING" ? "cal-badge--running" :
                  c.status === "UPCOMING" ? "cal-badge--upcoming" :

                  <article key={c.id} className="cal-contest-card">
                    <div className="cal-contest-date">
                      <span>{start.toLocaleDateString("zh-CN", { month: "short", day: "numeric" })}</span>

                    <div className="cal-contest-body">

                    </div>
                    <div className="cal-contest-meta">
                      <span className={`cal-badge ${statusClass}`}>{contestStatusLabel(c.status)}</span>
                      {c.url ? (
                        <a href={c.url} target="_blank" rel="noreferrer" className="ghost-button" style={{ fontSize: "12px" }}>
                          打开
                        </a>
                      ) : null}
                    </div>
                  </article>
                );
              })}
            </div>
          </div>
        ))}
      </section>
    </div>
  );
}

// ─── 折线图（提交趋势）────────────────────────────────
function LineChart({ data, width = 400, height = 120 }) {
  if (!data || data.length === 0) return <p className="muted">暂无数据</p>;
  const totals = data.map((d) => d.total);
  const maxVal = Math.max(...totals, 1);
  const pts = data.map((d, i) => {
    const x = (i / (data.length - 1 || 1)) * (width - 40) + 20;
    const y = height - 20 - ((d.total / maxVal) * (height - 30));
    return `${x},${y}`;
  });
  const acPts = data.map((d, i) => {
    const x = (i / (data.length - 1 || 1)) * (width - 40) + 20;
    const y = height - 20 - ((d.acCount / maxVal) * (height - 30));
    return `${x},${y}`;
  });
  return (
    <svg viewBox={`0 0 ${width} ${height}`} className="stat-svg">
      <polyline points={pts.join(" ")} fill="none" stroke="var(--accent)" strokeWidth="2" />
      <polyline points={acPts.join(" ")} fill="none" stroke="#22c55e" strokeWidth="2" strokeDasharray="4 2" />
      {data.map((d, i) => {
        const x = (i / (data.length - 1 || 1)) * (width - 40) + 20;
        return (
          <text key={i} x={x} y={height - 4} textAnchor="middle" fontSize="9" fill="var(--text-muted)">
            {d.week?.slice(5) ?? ""}
          </text>
        );
      })}
    </svg>
  );
}

// ─── 柱状图（标签正确率）────────────────────────────────
function BarChart({ data, width = 400, height = 150 }) {
  if (!data || data.length === 0) return <p className="muted">暂无数据</p>;
  const barW = Math.floor((width - 20) / data.length) - 4;
  return (
    <svg viewBox={`0 0 ${width} ${height}`} className="stat-svg">
      {data.map((d, i) => {
        const rate = d.attempts > 0 ? d.acCount / d.attempts : 0;
        const barH = Math.max(4, rate * (height - 30));
        const x = 10 + i * ((width - 20) / data.length);
        const y = height - 20 - barH;
        const color = rate < 0.3 ? "#ef4444" : rate < 0.6 ? "#f97316" : "#22c55e";
        return (
          <g key={d.tag}>
            <rect x={x} y={y} width={barW} height={barH} fill={color} rx="2" />
            <text x={x + barW / 2} y={height - 6} textAnchor="middle" fontSize="8" fill="var(--text-muted)">
              {d.tag?.slice(0, 6)}
            </text>
          </g>
        );
      })}
    </svg>
  );
}

// ─── 热力图（每日复习打卡）────────────────────────────────
function HeatmapChart({ data }) {
  if (!data || data.length === 0) return <p className="muted">暂无打卡数据</p>;
  const countByDay = new Map(data.map((d) => [d.day, d.count]));
  const today = new Date();
  const cells = [];
  for (let i = 89; i >= 0; i--) {
    const d = new Date(today);
    d.setDate(today.getDate() - i);
    const key = d.toISOString().slice(0, 10);
    const count = countByDay.get(key) ?? 0;
    cells.push({ key, count, label: key });
  }
  const maxCount = Math.max(...cells.map((c) => c.count), 1);
  const cols = 13; // ~13 weeks
  return (
    <div className="heatmap-grid">
      {cells.map((cell) => {
        const intensity = cell.count / maxCount;
        const bg = cell.count === 0
          ? "var(--border)"
          : `rgba(34,197,94,${0.2 + intensity * 0.8})`;
        return (
          <div
            key={cell.key}
            className="heatmap-cell"
            style={{ background: bg }}
            title={`${cell.label}: ${cell.count} 次`}
          />
        );
      })}
    </div>
  );
}
